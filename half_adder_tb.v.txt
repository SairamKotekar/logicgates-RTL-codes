`timescale 1ns/1ps

module testbench;

reg A, B;
wire Sum,Carry;

half_adder uut (
    .A(A),
    .B(B),
    .Sum(Sum),
    .Carry(Carry) 
);

initial begin
    $dumpfile("dump.vcd");
    $dumpvars(0, testbench);

  $monitor("A=%b B=%b Sum=%b Carry=%b", A, B, Sum, Carry);

    A=0; B=0; #10;
    A=0; B=1; #10;
    A=1; B=0; #10;
    A=1; B=1; #10;

    $finish;
end

endmodule
